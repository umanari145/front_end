flex-direction 各要素の並び方

右に並ぶ(初期値) flex-direction:row;

list1 row
list2 row
list3 row

逆順 flex-direction:row-reverse;

list1 row-reverse
list2 row-reverse
list3 row-reverse

縦並び flex-direction:column;

list1 column
list2 column
list3 column

逆縦並び flex-direction: column-reverse;

list1 column-reverse
list2 column-reverse
list3 column-reverse

justify-content 横の並び方

左づめ(初期値) justify-content:flex-start;

list1 left
list2 left
list3 left

右づめ justify-content:flex-end;

list1 right
list2 right
list3 right

中央揃え justify-content:center;

list1 center
list2 center
list3 center

均等ぞろえ(両端配置) justify-content:space-between;

list1 space-between
list2 space-between
list3 space-between

均等ぞろえ(両端スペースあり) justify-content:space-around;

list1 space-around
list2 space-around
list3 space-around

上ぞろえ align-items:flex-start;

list1 top
list2 top
list3 top

list1 bottom
list2 bottom
list3 bottom

中央ぞろえ align-items:center;

list2 middle
list3 middle

list1 baseline

list3 baseline

list1 strech

list3 strech

list2 bottom end
list3 bottom end

list3 middle center
